                    function getValues(){
                        //get values from the page

                        }
                        
                    
                       
                    }
                    
                    //generate numbers from startValue to the endValue
                    //logic function(s)
                    function generateNumbers(sValue, eValue){
                        
                        let numbers = [];
                    
                        //we want to get all numbers from start to end
                        for(let index = sValue; index <= eValue; index++){
                    
                            //this will execute in a loop until index = eValue
                    
                            numbers.push(index);
                        }
                    
                    
                    
                        return numbers;
                    }
                    
                    //display the numbers and mark even numbers bold
                    //display or view functions
                    function displayNumbers(numbers){
                    
                        let templateRows = "";
                    
                        for (let index = 0; index < numbers.length; index++) {
                    
                            let className = "even";
                    
                            let number = numbers[index];
                    
                            if(number % 2 == 0) {
                    
                                className = "even";
                    
                            } else {
                    
                                className = "odd";
                    
                            }
                            //This does render correctly with Prism. See the source.
                            templateRows += `${number}`;
                        }
                    
                        document.getElementById("results").innerHTML = templateRows;
                    }

Need for Speed

The function getValues is used to allow the user to put in a starting number and an ending number. The start and end values are parsed into integers to allow only numbers to be entered into the starting number and ending number boxes. An if statement is used to determine if a value is an integer. The starting and ending values are combined to make generateNumbers. The displayNumbers is responsible for showing the starting and ending numbers. Both generateNumbers and displayNumbers are called in the getValues function.

The function generateNumbers puts the numbers into an array beginning with the starting number and finishing with the ending number. A for loop is used to cycle through each of the numbers from the starting number to the ending number. The loop stops when it reaches the ending number that was entered. The numbers are then returned but not displayed at this moment.

The function displayNumbers has a for loop with an if statement that determines if the numbers are even or odd. The even numbers will print out in bold when the results are displayed. The document.getElementById is used to get the results of the table in rows with even numbers in bold.
